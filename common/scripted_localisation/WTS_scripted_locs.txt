defined_text = {
	name = WTS_upper_class_support
	text = {
		trigger = {
			check_variable = {

			}
		}
		localization_key = WTS_upper_class_support_maximum
	}
	text = {

			}
		}
		localization_key = WTS_upper_class_support_high
	}
	text = {
		trigger = {

				compare = less_than
			}
		}
		localization_key = WTS_upper_class_support_moderate
	}
	text = {
		trigger = {
			check_variable = {
				var = WTS.upper_class_support

				compare = greater_than_or_equals
			}
			check_variable = {
				var = WTS.upper_class_support
				value = 40
				compare = less_than
			}
		}
		localization_key = WTS_upper_class_support_low
	}
	text = {
		localization_key = WTS_upper_class_support_abyssmal
	}
}
