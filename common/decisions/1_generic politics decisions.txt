pro_yagoo_conservatism_on_the_rise = {
	open_up_political_discourse_pro_yagoo_conservative = {
		available = {
			always = yes
		}
		visible = {
			has_idea_with_trait = pro_yagoo_conservative_booster
			NOT = {
				has_government = democratic
			}
		}
		cost = 75
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			set_country_flag = pro_yagoo_conservative_discourse
		}
	}
	have_constitutional_referendum_pyc = {
		available = {
			democratic > 0.5
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_discourse
			NOT = {
				has_elections = yes
			}
		}
		cost = 100
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			add_stability = -0.1
			set_politics = {
				ruling_party = democratic
				elections_allowed = yes
				election_frequency = 48
			}
			country_event = politics.1
		}
	}
	oust_ruling_party_pyc = {
		available = {
			democratic > 0.5
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_discourse
			has_elections = yes
		}
		cost = 100
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			add_stability = -0.05
			set_politics = {
				ruling_party = democratic
				elections_allowed = yes
				election_frequency = 48
			}
			country_event = politics.2
		}
	}
	prepare_a_coup_pyc = {
		available = {
			democratic > 0.05
			command_power > 5
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_discourse
		}
		cost = 25
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			add_command_power = -5
			set_country_flag = pro_yagoo_conservative_coup
			activate_mission = upcoming_coup_pyc
			set_variable = {
				var = ROOT.coup_power
				value = 0
			}
		}
	}
	upcoming_coup_pyc = {
		allowed = {
			always = no
		}
		activation = {
			has_country_flag = pro_yagoo_conservative_coup
		}
		cancel_trigger = {
			has_government = democratic
		}
		days_mission_timeout = 90
		is_good = no
		selectable_mission = no
		#outcome of coup
		timeout_effect = {
			upcoming_coup_effect = yes
		}
	}
	bribe_important_figures_pyc = {
		available = {
			has_political_power > 50
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_coup
		}
		cost = 50
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			add_to_variable = {
				var = root.coup_power
				value = 2
				tooltip = GEN_coup_power_var_add_tooltip
			}
		}
	}
	facilitate_public_support_pyc = {
		available = {
			has_political_power > 50
			has_equipment = {
				infantry_equipment > 100
			}
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_coup
		}
		cost = 50
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			add_stability = -0.05
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = -100
			}
			add_to_variable = {
				var = root.coup_power
				value = 2
				tooltip = GEN_coup_power_var_add_tooltip
			}
		}
	}
	secure_weapons_pyc = {
		available = {
			has_equipment = {
				infantry_equipment > 50
			}
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_coup
		}
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = -50
			}
			add_to_variable = {
				var = root.coup_power
				value = 1
				tooltip = GEN_coup_power_var_add_tooltip
			}
		}
	}
	create_several_fail_safes_pyc = {
		available = {
			command_power > 5
			has_army_experience > 5
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_coup
		}
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			add_command_power = -5
			army_experience = -5
			add_to_variable = {
				var = root.coup_power
				value = 1
				tooltip = GEN_coup_power_var_add_tooltip
			}
		}
	}
	slander_ruling_party_pyc = {
		available = {
			has_political_power > 75
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_discourse
		}
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			add_popularity = {
				ideology = var:ROOT.current_party_ideology_group
				popularity = -0.1
			}
			add_popularity = {
				ideology = democratic
				popularity = 0.07
			}
		}
	}
	promote_ideology_pyc = {
		available = {
			has_political_power > 50
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_discourse
			NOT = {
				has_idea = ideology_promoted_pyc
			}
		}
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			add_timed_idea = {
				idea = ideology_promoted_pyc
				days = 180
			}
		}
	}
	prepare_for_civil_war_pyc = {
		available = {
			democratic > 0.2
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_discourse
		}
		cost = 25
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			set_country_flag = pro_yagoo_conservative_civil_war
			set_variable = {
				var = ROOT.civil_war_power
				value = 0.2
			}
			clamp_variable = {
				var = ROOT.civil_war_power
				min = 0
				max = 1
			}
			activate_mission = upcoming_civil_war_pyc
		}
	}
	upcoming_civil_war_pyc = {
		allowed = {
			always = no
		}
		activation = {
			has_country_flag = pro_yagoo_conservative_civil_war
		}
		cancel_trigger = {
			has_government = democratic
		}
		days_mission_timeout = 155
		is_good = no
		selectable_mission = no
		#outcome of coup
		timeout_effect = {
			start_civil_war = {
				ideology = var:ROOT.current_party_ideology_group
				ruling_party = democratic
				size = var:ROOT.civil_war_power
				keep_political_leader = yes
			}
			if = {
				limit = {
					has_country_flag = politics_paramilitaries_prepared
				}
				create_loyalist_militia = yes
			}
			every_country = {
				limit = {
					has_global_flag = holo_realistic_names
					has_war_with = ROOT
					is_dynamic_country = yes
				}
				meta_effect = {
					text = {
						set_cosmetic_tag = [get_tag_realism]_b
					}
					get_tag_realism = "[realism_dynamic_set_name]"
				}
			}
		}
	}
	prepare_paramilitaries_pyc = {
		available = {
			has_equipment = {
				infantry_equipment > 300
			}
		}
		visible = {
			has_country_flag = pro_yagoo_conservative_civil_war
		}
		cost = 25
		fire_only_once = yes
		ai_will_do = {
			factor = 5
		}
		complete_effect = {
			add_to_variable = {
				var = root.civil_war_power
				value = 0.15
				tooltip = GEN_civil_war_power_var_add_tooltip
			}

		}
	}
