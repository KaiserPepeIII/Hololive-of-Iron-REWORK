	id = watson_crackdown.1
	title = watson_crackdown.1.t
	desc = watson_crackdown.1.desc
	is_triggered_only = yes
	#Success
	option = {
		name = watson_crackdown.1.a
		trigger = {
			has_country_flag = WTS_successful_negotiations
		}
		add_political_power = 25
		add_stability = 0.1
		set_temp_variable = {
			var = pie_chart_change
			value = 10
		}
		WTS_wales_adding_government = yes
		every_state = {
			limit = {
				is_in_array = {
					array = WTS.counter_insurgency_wales_array
					value = THIS.id
				}
			}
			add_dynamic_modifier = {
				modifier = WTS_successful_crackdown
				days = 90
			}
		}
		clr_country_flag = WTS_successful_negotiations
	}
	#Tension Eased
	option = {
		name = watson_crackdown.1.b
		trigger = {
			has_country_flag = WTS_eased_tensions
		}
		add_political_power = 25
		add_stability = 0.05
		every_state = {
			limit = {
				is_in_array = {
					array = WTS.counter_insurgency_wales_array
					value = THIS.id
				}
			}
			add_dynamic_modifier = {
				modifier = WTS_successful_crackdown
				days = 60
			}
		}
		clr_country_flag = WTS_eased_tensions
	}
	#Critical Failure

		name = watson_crackdown.1.c
		trigger = {
			has_country_flag = WTS_critical_failure
		}
